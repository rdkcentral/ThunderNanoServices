cmake_minimum_required(VERSION 3.3)

project(YangInterfaces)

find_package(Thunder)
find_package(${NAMESPACE}Core REQUIRED)
find_package(CompileSettingsDebug REQUIRED)

find_package(ProxyStubGenerator REQUIRED)
find_package(JsonGenerator REQUIRED)

file(GLOB INTERFACE_HEADERS "I*.h")

ProxyStubGenerator(INPUT ${INTERFACE_HEADERS})
JsonGenerator(CODE INPUT ${INTERFACE_HEADERS})

file(GLOB PROXY_STUB_SOURCES "ProxyStubs*.cpp")
file(GLOB JSON_ENUM_SOURCES "JsonEnum*.cpp")
file(GLOB JSON_HEADERS "J*.h")

set(PLUGIN_YANG_MARSHALLING ${MODULE_NAME}Marshalling CACHE INTERNAL "")
set(PLUGIN_YANG_DEFINITIONS ${MODULE_NAME}Definitions CACHE INTERNAL "")

add_library(${PLUGIN_YANG_MARSHALLING} SHARED
    ${PROXY_STUB_SOURCES}
    Module.cpp)

target_link_libraries(${PLUGIN_YANG_MARSHALLING}
    PRIVATE
        ${NAMESPACE}Core::${NAMESPACE}Core
        CompileSettingsDebug::CompileSettingsDebug)

add_library(${PLUGIN_YANG_DEFINITIONS} SHARED
    ${JSON_ENUM_SOURCES}
    Module.cpp)

target_include_directories(${PLUGIN_YANG_DEFINITIONS} SYSTEM
    PRIVATE
        "..")

target_link_libraries(${PLUGIN_YANG_DEFINITIONS}
    PRIVATE
        ${NAMESPACE}Core::${NAMESPACE}Core
        CompileSettingsDebug::CompileSettingsDebug)

install(TARGETS ${PLUGIN_YANG_MARSHALLING} DESTINATION ../${EXAMPLE_DYNAMICLOADING_TARGET}/usr/${CMAKE_INSTALL_LIBDIR}/${STORAGE_DIRECTORY}/proxystubs)
install(TARGETS ${PLUGIN_YANG_DEFINITIONS} DESTINATION ../${EXAMPLE_DYNAMICLOADING_TARGET}/usr/${CMAKE_INSTALL_LIBDIR})
